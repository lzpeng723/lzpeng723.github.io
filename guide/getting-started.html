# Quick Start

# back-end

# clone the project
git clone https://github.com/lzpeng723/lzpeng-boot-server.git
# Modify the database configuration information in  src/main/resources/application-mysql.yml
# open the project directory
cd lzpeng-boot-server
# start the project
mvn springboot:run

# front-end

# clone the project
git clone https://github.com/lzpeng723/lzpeng-boot-client.git
# open the project directory
cd lzpeng-boot-client
# install dependencies
npm i -S
# start the project
npm run dev
